	@ {
		range: [0-100]
	}
	int number

	Slot targetSlot

	@ {
		precision: 1/60,
		range: <0.5, 1],
	}
	@ das
	@ dsa
	float time

	@ {
		onlyIf: targetSlot == Slot.First,
		atb2,
		%C#: CSharpOnlyAttribute,
		%go: {
			firstGoOnlyAttribute: 42,
			secondGoOnlyAttribute,
		},
	}
	Color championColor
}

	First,
	Second,
	Third,
}
